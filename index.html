        function tukarTab(idTab) {

            // Sembunyikan semua tab

            document.querySelectorAll('.tab-content').forEach(el => el.classList.add('hidden'));

                el.classList.remove('text-blue-600', 'border-blue-600');

                el.classList.add('text-gray-500', 'border-transparent');

            });


            // Tunjuk tab yang dipilih

            document.getElementById('tab-' + idTab).classList.remove('hidden');

            

            // Highlight butang (kecuali untuk keputusan)

            if(navBtn) {

                navBtn.classList.remove('text-gray-500', 'border-transparent');

            }

            if (idTab === 'imbas') {

                mulakanKamera();

            }

        }

                let divSoalan = document.createElement('div');

                divSoalan.className = "flex items-center justify-between bg-white p-3 rounded border shadow-sm";

                

                let htmlRadio = PILIHAN.map(p => `

                    <label class="flex items-center space-x-1 cursor-pointer">

                        <input type="radio" name="soalan_${i}" value="${p}" class="w-5 h-5 text-blue-600 focus:ring-blue-500" ${skemaJawapan[i] === p ? 'checked' : ''} onchange="setJawapan(${i}, '${p}')">

                        <span class="font-medium text-gray-700">${p}</span>

                    </label>

                `).join('');


                divSoalan.innerHTML = `

                    <div class="font-bold text-gray-700 w-8">${i + 1}.</div>

                    <div class="flex space-x-4">${htmlRadio}</div>

                `;

                bekas.appendChild(divSoalan);

            }

        }

        // ==========================================

        async function mulakanKamera() {

            const video = document.getElementById('kamera');

            if (streamKamera) return; // Dah jalan


            try {

                // Minta kamera belakang (environment)

                streamKamera = await navigator.mediaDevices.getUserMedia({ 

                    video: { facingMode: 'environment', width: { ideal: 1280 }, height: { ideal: 720 } } 

                });

                video.srcObject = streamKamera;

            } catch (err) {

                console.error("Ralat kamera:", err);

                alert("Tak dapat buka kamera. Pastikan cikgu benarkan akses kamera untuk laman web ni.");

            }

        }


        function hentikanKamera() {

            if (streamKamera) {

                streamKamera.getTracks().forEach(track => track.stop());

                streamKamera = null;

                document.getElementById('kamera').srcObject = null;

            }

        }


        function tangkapDanTanda() {

            if (skemaJawapan.includes(null)) {

                alert("Cikgu belum siapkan skema jawapan! Sila ke tab 'Skema' dahulu.");

                return;

            }


            const video = document.getElementById('kamera');

            if (!video.videoWidth) {

                alert("Kamera belum bersedia.");

                return;

            }


            // Buat canvas sementara untuk tangkap imej penuh dari video

            const canvas = document.createElement('canvas');

            canvas.width = video.videoWidth;

            canvas.height = video.videoHeight;

            const ctx = canvas.getContext('2d', { willReadFrequently: true });

            ctx.drawImage(video, 0, 0, canvas.width, canvas.height);


            // Analisis Imej

            analisisImej(canvas);

        }


        function analisisImej(sumberCanvas) {

            const ctx = sumberCanvas.getContext('2d');

            const cw = sumberCanvas.width;

            const ch = sumberCanvas.height;


            // Kita assume kotak panduan merah adalah pada: top: 10%, left: 10%, w: 80%, h: 80% (rujuk CSS style)

            // Ini bermaksud kertas OMR patut berada dalam kawasan piksel ini:

            const boxX = cw * 0.10;

            const boxY = ch * 0.10;

            const boxW = cw * 0.80;

            const boxH = ch * 0.80;


            // Susunan geografi borang cetakan kita (secara peratusan kasar dari kotak borang)

            // Ketinggian Header (Nama dll) lebih kurang 20% dari atas

            const headerOffsetP = 0.22; 

            const ruangSoalanH = boxH * (1 - headerOffsetP);

            const mulaY = boxY + (boxH * headerOffsetP);

            

            const jarakBaris = ruangSoalanH / JUMLAH_SOALAN;

            

            // X offset untuk A, B, C, D (berdasarkan rupa grid cetak tadi)

            // Kotak nombor ambil ~15% ruang kiri, border dash. Pilihan bermula dari ~25% hingga 90%

            const posXPilihan = [

                boxX + boxW * 0.30, // A

                boxX + boxW * 0.50, // B

                boxX + boxW * 0.70, // C

                boxX + boxW * 0.90  // D

            ];


            let markah = 0;

            let butiran = [];

            let markahDikesan = [];


            // Sediakan debug canvas untuk tunjuk apa yang sistem nampak

            const canvasDebug = document.getElementById('canvas-debug');

            canvasDebug.width = cw;

            canvasDebug.height = ch;

            const ctxDebug = canvasDebug.getContext('2d');

            ctxDebug.drawImage(sumberCanvas, 0, 0);

            

            // Lukis kotak tekaan

            ctxDebug.strokeStyle = "yellow";

            ctxDebug.lineWidth = 4;

            ctxDebug.strokeRect(boxX, boxY, boxW, boxH);


            // Loop setiap soalan dan pilihan untuk periksa kegelapan piksel

            for (let i = 0; i < JUMLAH_SOALAN; i++) {

                let cy = mulaY + (i * jarakBaris) + (jarakBaris / 2);

                let tahapKegelapan = [];

                let pilihanPelajar = null;

                

                for (let j = 0; j < PILIHAN.length; j++) {

                    let cx = posXPilihan[j];

                    

                    // Saiz kawasan pensampelan (sampling area)

                    let r = boxW * 0.02; // 2% lebar kotak

                    

                    // Dapatkan data piksel dalam bulatan kecil di (cx, cy)

                    let imgData = ctx.getImageData(cx - r, cy - r, r*2, r*2);

                    let pixels = imgData.data;

                    let totalBrightness = 0;

                    

                    // Kira purata kecerahan (brightness)

                    for (let p = 0; p < pixels.length; p += 4) {

                        let red = pixels[p];

                        let green = pixels[p+1];

                        let blue = pixels[p+2];

                        // Formula luminan standard

                        let brightness = (0.299 * red + 0.587 * green + 0.114 * blue);

                        totalBrightness += brightness;

                    }

                    let avgBrightness = totalBrightness / (pixels.length / 4);

                    tahapKegelapan.push(avgBrightness);


                    // Debug: Lukis titik ujian di canvas

                    ctxDebug.beginPath();

                    ctxDebug.arc(cx, cy, r, 0, 2 * Math.PI);

                    // Kalau gelap (nilai brightness rendah, let say < 130), kaler hijau

                    if (avgBrightness < 130) {

                        ctxDebug.fillStyle = "rgba(0, 255, 0, 0.5)";

                    } else {

                        ctxDebug.fillStyle = "rgba(255, 0, 0, 0.3)";

                    }

                    ctxDebug.fill();

                    ctxDebug.stroke();

                }


                // Cari bulatan yang paling gelap

                let nilaiPalingGelap = Math.min(...tahapKegelapan);

                

                // Threshold: adakah ia cukup gelap untuk dikira sebagai "dihitamkan"?

                // Jika kurang dari 130 (0=hitam pekat, 255=putih bersih), kita anggap ditanda.

                if (nilaiPalingGelap < 130) {

                    let indeksJawapan = tahapKegelapan.indexOf(nilaiPalingGelap);

                    pilihanPelajar = PILIHAN[indeksJawapan];

                    

                    // Semak jika ada lebih dari 1 bulatan yang dihitamkan (double mark)

                    let countGelap = tahapKegelapan.filter(b => b < 130).length;

                    if (countGelap > 1) {

                        pilihanPelajar = "BATAL"; // Tandakan sebagai batal

                    }

                }


                // Semak betul atau salah

                let betul = false;

                let jawapanSebenar = skemaJawapan[i];

                if (pilihanPelajar === jawapanSebenar) {

                    betul = true;

                    markah++;

                }


                butiran.push({

                    soalan: i + 1,

                    jawapanPelajar: pilihanPelajar || 'KOSONG',

                    jawapanSebenar: jawapanSebenar,

                    betul: betul

                });

            }


            paparKeputusan(markah, butiran);

        }


        function paparKeputusan(markah, butiran) {

            tukarTab('keputusan'); // Tukar ke tab hasil, ini akan tutup kamera automatik


            let peratus = (markah / JUMLAH_SOALAN) * 100;

            document.getElementById('skor-markah').innerText = `${markah}/${JUMLAH_SOALAN}`;

            document.getElementById('skor-peratus').innerText = `${peratus.toFixed(0)}%`;


            let senaraiHtml = butiran.map(b => {

                let ikon = b.betul ? '✅' : '❌';

                let warnaTeks = b.betul ? 'text-green-600 font-bold' : 'text-red-600 font-bold';

                return `

                    <div class="flex justify-between items-center p-2 bg-white rounded border">

                        <div class="w-16">No. ${b.soalan}</div>

                        <div class="flex-1 text-center">Jawab: <span class="${warnaTeks}">${b.jawapanPelajar}</span></div>

                        <div class="flex-1 text-right text-gray-500">Skema: ${b.jawapanSebenar} ${ikon}</div>

                    </div>

                `;

            }).join('');


            document.getElementById('senarai-keputusan').innerHTML = senaraiHtml;

        }
